



import { Link } from "react-router-dom";
import { useState } from "react";
import ArrowUp from "../../assets/images/nav/Arrow.png";

export default function Footer() {
  const [isFooterOpen, setIsFooterOpen] = useState(false);

  const toggleFooter = () => {
    setIsFooterOpen(!isFooterOpen);
  };

  return (
    <>
      {/* Footer Top */}
      <div className="w-full h-11 text-black bg-red-600 justify-center items-center inline-flex">
        {/* Your Footer Top Content Goes Here */}
      </div>

      {/* Footer Content */}
      <footer className="bg-violet-800">
        {/* Arrow Button for Toggling Footer */}
        <div className="inline-flex md:hidden flex-end pe-1">
          <button
            onClick={toggleFooter}
            title="Toggle Footer"
            type="button"
            className="inline-flex items-center text-sm text-gray-500 md:hidden"
            aria-expanded={isFooterOpen ? "true" : "false"}
          >
            <img src={ArrowUp} alt="Arrow Up" />
          </button>
        </div>

        {/* Footer Content Block */}
        <div
          className={`${isFooterOpen ? "block" : "hidden"} w-full md:w-auto lg:w-auto`}
          id="footer-content"
        >
          {/* Your Footer Content Goes Here */}
        </div>
      </footer>
    </>
  );
}
